58  inline void alignChildren() {
59  const float innerRectSize = m_scale.y;
60 
61  // Available space -= margin
62  float sizeUsed = m_margin;
63 
64  // Available space -= the dimensions of fixed-sized elements
65  int fixedElementCount = 0;
66  for each (const auto & pair in m_sizedChildren) {
67  auto[child, ratio] = pair;
68  if (!std::isnan(child->getMinScale().y) || !std::isnan(child->getMaxScale().y)) {
69  sizeUsed += child->getScale().y * ratio;
70  fixedElementCount++;
71  }
72  }
73 
74  // Available space -= spacing factor between elements
75  if (m_sizedChildren.size() > 1)
76  sizeUsed += float(m_sizedChildren.size() - 1ull) * m_spacing;
77 
78  // Remaining space divvied up between remaining elements
79  const float remainder = innerRectSize - sizeUsed;
80  const float elementSize = remainder / (float(m_sizedChildren.size()) - float(fixedElementCount));
81 
82  float positionFromTop = m_scale.y - m_margin;
83  const float top = positionFromTop;
84  for each (const auto & pair in m_sizedChildren) {
85  auto[child, ratio] = pair;
86  child->setScale(glm::vec2(m_scale.x - m_margin, elementSize * ratio));
87  if (m_sizedChildren.size() == 1) {
88  child->setPosition(glm::vec2(0.0f));
89  continue;
90  }
91  const float size = child->getScale().y;
92  positionFromTop -= size;
93  child->setPosition(glm::vec2(0, positionFromTop));
94  positionFromTop -= size + (m_spacing * 2.0f);
95  }
96  const float bottom = positionFromTop + (m_spacing * 2.0f);
97 
98  // Edge Case: all elements are fixed size, gap may be present
99  // Solution: change spacing to fit all elements within bounds
100  if (m_sizedChildren.size() > 1 && fixedElementCount == m_sizedChildren.size()) {
101  const float delta = (bottom - top) / float(m_sizedChildren.size() + size_t(1));
102 
103  for (size_t x = 1; x < m_sizedChildren.size(); ++x)
104  m_sizedChildren[x].first->setPosition(m_sizedChildren[x].first->getPosition() + glm::vec2(0, delta * x));
105  }
106  }
