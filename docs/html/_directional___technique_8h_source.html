84  inline virtual void updateTechnique(const float & deltaTime) override {
85  // Link together the dimensions of view info to that of the viewport vectors
86  m_frameData->viewInfo.resize(m_cameras->size());
87 
88  // Render important shadows
89  if (m_enabled && m_frameData->viewInfo.size() && m_shader_Bounce->existsYet()) {
90  // Find client camera
91  size_t visibilityIndex = 0;
92  bool found = false;
93  for (size_t x = 0; x < m_cameras->size(); ++x)
94  if (m_cameras->at(x) == m_frameData->clientCamera.get()) {
95  visibilityIndex = x;
96  found = true;
97  break;
98  }
99  if (found && visibilityIndex < m_frameData->viewInfo.size() && m_frameData->viewInfo[visibilityIndex].visShadowCount) {
100  // Light bounce using client camera
101  if (m_bounceIndex >= m_drawBounceData.size())
102  m_drawBounceData.resize(size_t(m_drawIndex) + 1ull);
103 
104  auto & bounceBuffer = m_drawBounceData[m_bounceIndex];
105  auto &camBufferIndex = bounceBuffer.bufferCamIndex;
106  auto &lightBufferIndex = bounceBuffer.visLights;
107  camBufferIndex.beginWriting();
108  lightBufferIndex.beginWriting();
109  bounceBuffer.indirectBounce.beginWriting();
110 
111  // Accumulate all visibility info for the cameras passed in
112  std::vector<glm::ivec2> camIndices(m_frameData->viewInfo[visibilityIndex].lightIndices.size(), { visibilityIndex, 0 });
113  std::vector<GLint> lightIndices(m_frameData->viewInfo[visibilityIndex].lightIndices.begin(), m_frameData->viewInfo[visibilityIndex].lightIndices.end());
114  const auto shadowCount = m_frameData->viewInfo[visibilityIndex].visShadowCount;
115 
116  if (lightIndices.size()) {
117  // Write accumulated data
118  camBufferIndex.write(0, sizeof(glm::ivec2) * camIndices.size(), camIndices.data());
119  lightBufferIndex.write(0, sizeof(GLuint) * lightIndices.size(), lightIndices.data());
120  const GLuint dataBounce[] = { (GLuint)m_shapeQuad->getSize(), (GLuint)(shadowCount * m_bounceSize),0,0 };
121  bounceBuffer.indirectBounce.write(0, sizeof(GLuint) * 4, &dataBounce);
122 
123  // Update light bounce
124  renderBounce(deltaTime, (int)visibilityIndex);
125 
126  m_bounceIndex++;
127  }
128  }
129  }

132  // Exit Early

134  if (m_drawIndex >= m_drawData.size())
135  m_drawData.resize(size_t(m_drawIndex) + 1ull);
136  auto & drawBuffer = m_drawData[m_drawIndex];
137  auto &camBufferIndex = drawBuffer.bufferCamIndex;
138  auto &lightBufferIndex = drawBuffer.visLights;
139  camBufferIndex.beginWriting();
140  lightBufferIndex.beginWriting();
141  drawBuffer.indirectShape.beginWriting();
142 
143  // Accumulate all visibility info for the cameras passed in
144  std::vector<glm::ivec2> camIndices;
145  std::vector<GLint> lightIndices;
146  for (auto &[camIndex, layer] : perspectives) {
147  const std::vector<glm::ivec2> tempIndices(m_frameData->viewInfo[camIndex].lightIndices.size(), { camIndex, layer });
148  camIndices.insert(camIndices.end(), tempIndices.begin(), tempIndices.end());
149  lightIndices.insert(lightIndices.end(), m_frameData->viewInfo[camIndex].lightIndices.begin(), m_frameData->viewInfo[camIndex].lightIndices.end());
150  }
151 
152  if (lightIndices.size()) {
153  // Write accumulated data
154  camBufferIndex.write(0, sizeof(glm::ivec2) * camIndices.size(), camIndices.data());
155  lightBufferIndex.write(0, sizeof(GLuint) * lightIndices.size(), lightIndices.data());
156  const GLuint dataShape[] = { (GLuint)m_shapeQuad->getSize(), (GLuint)lightIndices.size(), 0,0 };
157  drawBuffer.indirectShape.write(0, sizeof(GLuint) * 4, &dataShape);
158 
159  // Render lights
160  renderLights(deltaTime, viewport);
161 
162  m_drawIndex++;
163  }
