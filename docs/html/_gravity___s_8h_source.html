39  for (unsigned int y = 2u; y < 12u; ++y)
40  for (unsigned int x = 0u; x < 6u; ++x) {
41  const auto & xTile = board.m_tiles[y][x];
42  // Exclude any background tiles, scored tiles, or already falling tiles
43  if (xTile.m_type != TileState::NONE && xTile.m_scoreType == TileState::UNMATCHED && board.m_tileDrops[y][x].dropState == Board_Component::TileDropData::STATIONARY) {
44 
45  // Determine how far this tile may fall
46  unsigned int dropIndex = y;
47  while (dropIndex - 1u > 0 && board.m_tiles[dropIndex - 1u][x].m_type == TileState::NONE)
48  dropIndex--;
49  unsigned int dropDistance = y - dropIndex;
50  const auto & dTile = board.m_tiles[dropIndex][x];
51 
52  // Determine the amount of weight imposed on this tile
53  unsigned int weight = 0;
54  for (unsigned int z = y; z < 12u; ++z) {
55  if (board.m_tiles[z][x].m_scoreType == TileState::SCORED)
56  break;
57  if (board.m_tiles[z][x].m_type == TileState::NONE)

59  weight++;
60  }
61 
62  // Continue only if this tile WILL ACTUALLY FALL

65  for (unsigned int z = y; z < 12u; ++z) {
66  // Add a skip for dead space
67  if (board.m_tiles[z][x].m_type == TileState::NONE) {
68  dropDistance++;
69  continue;
70  }

73  break;

75  }
76  }
77  }
78  }

102  }

113  }
114  // Tile has already been dropped to destination, need to move tile upwards now, not down
115  board.m_data->data->tiles[(y * 6) + x].yOffset = 2.0f * (1.0f - (easeOutBounce(adjustedTime / bounceTime))) * bounceHeight;
116  break;
117  }
118  // Find Stationary Tiles
119  default: {
120  // Reset the data just in case
121  board.m_data->data->tiles[(y * 6) + x].yOffset = 0.0f;
122  board.m_tileDrops[y][x] = Board_Component::TileDropData();
123  break;
124  }
125  }
126  }
