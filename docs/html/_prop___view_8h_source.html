101  inline virtual void updateComponents(const float & deltaTime, const std::vector< std::vector<BaseECSComponent*> > & components) override {
102  // Accumulate draw parameter information per model
103  std::vector<glm::ivec4> cullingDrawData, renderingDrawData;
104  std::vector<GLuint> visibleIndices;
105  std::vector<int> skeletonData;
106  const glm::vec3 & eyePosition = (*m_cameraBuffer)->EyePosition;
107  for each (const auto & componentParam in components) {
108  Prop_Component * propComponent = (Prop_Component*)componentParam[0];
109  Skeleton_Component * skeletonComponent = (Skeleton_Component*)componentParam[1];
110  Transform_Component * transformComponent = (Transform_Component*)componentParam[2];
111  const auto & offset = propComponent->m_model->m_offset;
112  const auto & count = propComponent->m_model->m_count;
113  const auto & index = *propComponent->m_propBufferIndex;
114  if (!propComponent->m_model->existsYet()) continue;
115 
116  // Sync Transform Attributes
117  if (transformComponent) {
118  const auto & position = transformComponent->m_transform.m_position;
119  const auto & orientation = transformComponent->m_transform.m_orientation;
120  const auto & scale = transformComponent->m_transform.m_scale;
121  const auto matRot = glm::mat4_cast(orientation);
122  propComponent->mMatrix = transformComponent->m_transform.m_modelMatrix;
123 
124  // Update bounding sphere
125  const glm::vec3 bboxMax_World = (propComponent->m_model->m_bboxMax * scale) + position;
126  const glm::vec3 bboxMin_World = (propComponent->m_model->m_bboxMin * scale) + position;
127  const glm::vec3 bboxCenter = (bboxMax_World + bboxMin_World) / 2.0f;
128  const glm::vec3 bboxScale = (bboxMax_World - bboxMin_World) / 2.0f;
129  glm::mat4 matTrans = glm::translate(glm::mat4(1.0f), bboxCenter);
130  glm::mat4 matScale = glm::scale(glm::mat4(1.0f), bboxScale);
131  glm::mat4 matFinal = (matTrans * matRot * matScale);
132  propComponent->bBoxMatrix = matFinal;
133  propComponent->m_radius = glm::compMax(propComponent->m_model->m_radius * scale);
134  propComponent->m_position = propComponent->m_model->m_bboxCenter + position;
135 
136  }
137 
138  // Sync Animation Attributes
139  if (skeletonComponent) {
140  auto & bones = m_skeletonBuffer[*skeletonComponent->m_skeleBufferIndex].bones;
141  for (size_t i = 0, total = std::min(skeletonComponent->m_transforms.size(), size_t(NUM_MAX_BONES)); i < total; ++i)
142  bones[i] = skeletonComponent->m_transforms[i];
143  }
144 
145  // Sync Prop Attributes
146  m_propBuffer[index].materialID = propComponent->m_skin;
147  m_propBuffer[index].mMatrix = propComponent->mMatrix;
148  m_propBuffer[index].bBoxMatrix = propComponent->bBoxMatrix;
149 
150  // Flag for occlusion culling if mesh complexity is high enough and if viewer is NOT within BSphere
151  visibleIndices.push_back((GLuint)index);
152  if ((count >= 100) && propComponent->m_radius < glm::distance(propComponent->m_position, eyePosition)) {
153  // Allow
154  cullingDrawData.push_back(glm::ivec4(36, 1, 0, 1));
155  renderingDrawData.push_back(glm::ivec4(count, 0, offset, 1));
156  }
157  else {
158  // Skip occlusion culling
159  cullingDrawData.push_back(glm::ivec4(36, 0, 0, 1));
160  renderingDrawData.push_back(glm::ivec4(count, 1, offset, 1));
161  }
162  skeletonData.push_back(skeletonComponent ? (GLint)*skeletonComponent->m_skeleBufferIndex : -1); // get skeleton ID if this entity has one
163  }
164 
165  // Update camera buffers
166  m_propCount = (GLsizei)visibleIndices.size();
167  m_bufferPropIndex.write(0, sizeof(GLuint) * m_propCount, visibleIndices.data());
168  m_bufferCulling.write(0, sizeof(glm::ivec4) * m_propCount, cullingDrawData.data());
