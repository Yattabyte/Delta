41  // Accumulate Light Data
42  auto & graphics = m_engine->getModule_Graphics();
43  const auto cameraBuffer = graphics.getActiveCameraBuffer();
44  const glm::vec2 &size = cameraBuffer->data->Dimensions;
45  const float ar = size.x / size.y;
46  const float tanHalfHFOV = glm::radians(cameraBuffer->data->FOV) / 2.0f;
47  const float tanHalfVFOV = atanf(tanf(tanHalfHFOV) / ar);
48  const float near_plane = -CAMERA_NEAR_PLANE;
49  const float far_plane = -m_renderState.m_drawDistance/2.0F;
50  float cascadeEnd[NUM_CASCADES + 1];
51  glm::vec3 middle[NUM_CASCADES], aabb[NUM_CASCADES];
52  constexpr float lambda = 0.75f;
53  cascadeEnd[0] = near_plane;
54  for (int x = 1; x < NUM_CASCADES + 1; ++x) {
55  const float xDivM = float(x) / float(NUM_CASCADES);
56  const float cLog = near_plane * powf((far_plane / near_plane), xDivM);
57  const float cUni = near_plane + (far_plane - near_plane) * xDivM;
58  cascadeEnd[x] = (lambda * cLog) + (1.0f - lambda) * cUni;
59  }
60  for (int i = 0; i < NUM_CASCADES; i++) {
61  float points[4] = {
62  cascadeEnd[i] * tanHalfHFOV,
63  cascadeEnd[i + 1] * tanHalfHFOV,
64  cascadeEnd[i] * tanHalfVFOV,
65  cascadeEnd[i + 1] * tanHalfVFOV
66  };
67  float maxCoord = std::max(abs(cascadeEnd[i]), abs(cascadeEnd[i + 1]));
68  for (int x = 0; x < 4; ++x)
69  maxCoord = std::max(maxCoord, abs(points[x]));
70  // Find the middle of current view frustum chunk
71  middle[i] = glm::vec3(0, 0, ((cascadeEnd[i + 1] - cascadeEnd[i]) / 2.0f) + cascadeEnd[i]);
72 
73  // Measure distance from middle to the furthest point of frustum slice
74  // Use to make a bounding sphere, but then convert into a bounding box
75  aabb[i] = glm::vec3(glm::distance(glm::vec3(maxCoord), middle[i]));
76  }
77  const glm::mat4 CamInv = glm::inverse(cameraBuffer->data->vMatrix);
78  const glm::mat4 CamP = cameraBuffer->data->pMatrix;
79  std::vector<GLint> lightIndices, shadowIndices;
80  PriorityList<float, std::pair<LightDirectional_Component*, LightDirectionalShadow_Component*>, std::less<float>> oldest;
81  for each (const auto & componentParam in components) {
82  LightDirectional_Component * lightComponent = (LightDirectional_Component*)componentParam[0];
83  LightDirectionalShadow_Component * shadowComponent = (LightDirectionalShadow_Component*)componentParam[1];
84  lightIndices.push_back(lightComponent->m_data->index);
85  if (shadowComponent) {
86  shadowIndices.push_back(shadowComponent->m_data->index);
87  oldest.insert(shadowComponent->m_updateTime, std::make_pair(lightComponent, shadowComponent));
88  for (int i = 0; i < NUM_CASCADES; i++) {
89  const glm::vec3 volumeUnitSize = (aabb[i] - -aabb[i]) / (float)m_renderState.m_shadowSize.x;
90  const glm::vec3 frustumpos = glm::vec3(shadowComponent->m_mMatrix * CamInv * glm::vec4(middle[i], 1.0f));
91  const glm::vec3 clampedPos = glm::floor((frustumpos + (volumeUnitSize / 2.0f)) / volumeUnitSize) * volumeUnitSize;
92  const glm::vec3 newMin = -aabb[i] + clampedPos;
93  const glm::vec3 newMax = aabb[i] + clampedPos;
94  const float l = newMin.x, r = newMax.x, b = newMax.y, t = newMin.y, n = -newMin.z, f = -newMax.z;
95  const glm::mat4 pMatrix = glm::ortho(l, r, b, t, n, f);
96  const glm::mat4 pvMatrix = pMatrix * shadowComponent->m_mMatrix;
97  shadowComponent->m_data->data->lightVP[i] = pvMatrix;
98  shadowComponent->m_data->data->inverseVP[i] = inverse(pvMatrix);
99  const glm::vec4 v1 = glm::vec4(0, 0, cascadeEnd[i + 1], 1.0f);
100  const glm::vec4 v2 = CamP * v1;
101  shadowComponent->m_data->data->CascadeEndClipSpace[i] = v2.z;
102  }
103  }
104  else
105  shadowIndices.push_back(-1);
106  }
107 
108  // Update Draw Buffers
109  const size_t & lightSize = lightIndices.size(), & shadowSize = shadowIndices.size();
110  const GLuint bounceInstanceCount = GLuint(shadowSize * m_renderState.m_bounceSize);
111  m_renderState.m_visLights.write(0, sizeof(GLuint) * lightSize, lightIndices.data());
112  m_renderState.m_visShadows.write(0, sizeof(GLuint) * shadowSize, shadowIndices.data());
113  m_renderState.m_indirectShape.write(sizeof(GLuint), sizeof(GLuint), &lightSize); // update primCount (2nd param)
114  m_renderState.m_indirectBounce.write(sizeof(GLuint), sizeof(GLuint), &bounceInstanceCount);
115  m_renderState.m_shadowsToUpdate = PQtoVector(oldest);
116  m_renderState.m_shadowCount = (GLint)shadowSize;
117  }
