84  inline virtual void mouseAction(const MouseEvent & mouseEvent) override {
85  UI_Element::mouseAction(mouseEvent);
86  if (getVisible() && getEnabled() && mouseWithin(mouseEvent)) {
87  // Move hover selection to whatever is beneath mouse
88  MouseEvent subEvent = mouseEvent;
89  subEvent.m_xPos = mouseEvent.m_xPos - m_position.x;
90  subEvent.m_yPos = mouseEvent.m_yPos - m_position.y;
91  int index(0);
92  for each (auto & child in m_children) {
93  if (child->mouseWithin(subEvent)) {
94  setHoverIndex(index); // Set hovered item to whatever is beneath mouse
95  if (mouseEvent.m_action == MouseEvent::RELEASE)
96  setSelectionIndex(index); // Set selected item to whatever is beneath mouse
97  break;
98  }
99  else
100  index++;
101  }
102 
103  // Force current selection to stay highlighted
104  if (m_children.size() && m_hoverIndex > -1)
105  m_children[m_hoverIndex]->setHovered();
106  }

108  inline virtual void userAction(ActionState & actionState) {
109  // User can go up or down the list_Horizontal with an input device
110  // User input wraps around, and if an item is selected, moving will deselect it
111  if (m_children.size()) {
112  // Allow selected child to receive input first
113  if (m_selectionIndex >= 0 && m_selectionIndex < m_children.size())
114  m_focusMap.applyActionState(actionState);
115 
116  // After, process remaining input for the list_Horizontal
117  if (actionState.isAction(ActionState::UI_UP) == ActionState::PRESS) {
118  setHoverIndex(m_hoverIndex - 1 < 0 ? (int)m_children.size() - 1ull : m_hoverIndex - 1);
119 
120  if (m_selectionIndex != -1)
121  setSelectionIndex(-1);
122  }
123  else if (actionState.isAction(ActionState::UI_DOWN) == ActionState::PRESS) {
124  setHoverIndex(m_hoverIndex + 1 > m_children.size() - 1ull ? 0 : m_hoverIndex + 1);
125 
126  if (m_selectionIndex != -1)
127  setSelectionIndex(-1);
128  }
129  else if (actionState.isAction(ActionState::UI_ENTER) == ActionState::PRESS) {
130  if (m_hoverIndex > -1 && m_hoverIndex < m_children.size())
131  setSelectionIndex(m_hoverIndex);
132  }
133  }
134  }
