52  const auto ray_eye = glm::vec4(glm::vec2(clientCamera.pMatrixInverse * glm::vec4(ray_nds, -1.0f, 1.0F)), -1.0f, 0.0f);
53  const auto ray_direction = glm::normalize(glm::vec3(clientCamera.vMatrixInverse * ray_eye));
54 
55  // Set the selection position for the worst-case scenario
56  m_selectionTransform.m_position = clientCamera.EyePosition + (ray_direction * glm::vec3(50.0f));
57  m_intersectionTransform.m_position = clientCamera.EyePosition + (ray_direction * glm::vec3(50.0f));

60  int highestConfidence = 0;
61  bool found = false;
62  for each (const auto & componentParam in components) {
63  auto * transformComponent = (Transform_Component*)componentParam[0];
64  auto * bBox = (BoundingBox_Component*)componentParam[1];
65  auto * bSphere = (BoundingSphere_Component*)componentParam[2];
66  auto * collider = (Collider_Component*)componentParam[3];
67  auto * prop = (Prop_Component*)componentParam[4];
68  float distanceFromScreen = FLT_MAX;
69  int confidence = 0;
70  const bool hasBoundingShape = bSphere || bBox;
71  bool result = false;
72 
73  // Attempt cheap tests first
74  result = RayOrigin(transformComponent, ray_origin, ray_direction, distanceFromScreen, confidence);
75  if (bBox)
76  result = RayBBox(transformComponent, bBox, ray_origin, ray_direction, distanceFromScreen, confidence);
77  else if (bSphere)
78  result = RayBSphere(transformComponent, bSphere, ray_origin, ray_direction, distanceFromScreen, confidence);
79 
80  // Attempt more complex tests
81  if (prop && ((hasBoundingShape && result) || (!hasBoundingShape))) {
82  distanceFromScreen = FLT_MAX;
83  result = RayProp(transformComponent, prop, ray_origin, ray_direction, distanceFromScreen, confidence);
84  }
85  else if (collider && ((hasBoundingShape && result) || (!hasBoundingShape))) {
86  distanceFromScreen = FLT_MAX;
87  result = RayCollider(transformComponent, collider, ray_origin, ray_direction, distanceFromScreen, confidence);
88  }
89 
90  // Find the closest best match

92  closestDistance = distanceFromScreen;
93  highestConfidence = confidence;

95  m_selectionTransform = transformComponent->m_worldTransform;
96  found = true;
97  }
98  }

101  m_intersectionTransform.update();
102  }
103  }
