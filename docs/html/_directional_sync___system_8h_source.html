64  const float tanHalfVFOV = atanf(tanf(tanHalfHFOV) / ar);

70  cascadeEnd[0] = near_plane;
71  for (int x = 1; x < NUM_CASCADES + 1; ++x) {
72  const float xDivM = float(x) / float(NUM_CASCADES);
73  const float cLog = near_plane * powf((far_plane / near_plane), xDivM);
74  const float cUni = near_plane + (far_plane - near_plane) * xDivM;

111  }
