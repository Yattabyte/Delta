51  const auto size = glm::vec2(m_frameData->clientCamera.get()->get()->Dimensions);
52  const float ar = size.x / size.y;
53  const float tanHalfHFOV = glm::radians(m_frameData->clientCamera.get()->get()->FOV) / 2.0f;
54  const float tanHalfVFOV = atanf(tanf(tanHalfHFOV) / ar);
55  const float near_plane = -Camera::ConstNearPlane;
56  const float far_plane = -m_frameData->clientCamera.get()->get()->FarPlane;
57  float cascadeEnd[NUM_CASCADES + 1];

59  constexpr float lambda = 0.75f;
60  cascadeEnd[0] = near_plane;
61  for (int x = 1; x < NUM_CASCADES + 1; ++x) {
62  const float xDivM = float(x) / float(NUM_CASCADES);
63  const float cLog = near_plane * powf((far_plane / near_plane), xDivM);
64  const float cUni = near_plane + (far_plane - near_plane) * xDivM;
65  cascadeEnd[x] = (lambda * cLog) + (1.0f - lambda) * cUni;
66  }
67  for (int i = 0; i < NUM_CASCADES; i++) {
68  float points[4] = {
69  cascadeEnd[i] * tanHalfHFOV,
70  cascadeEnd[i + 1] * tanHalfHFOV,
71  cascadeEnd[i] * tanHalfVFOV,
72  cascadeEnd[i + 1] * tanHalfVFOV
73  };
74  float maxCoord = std::max(abs(cascadeEnd[i]), abs(cascadeEnd[i + 1]));
75  for (int x = 0; x < 4; ++x)
76  maxCoord = std::max(maxCoord, abs(points[x]));
77  // Find the middle of current view frustum chunk
78  middle[i] = glm::vec3(0, 0, ((cascadeEnd[i + 1] - cascadeEnd[i]) / 2.0f) + cascadeEnd[i]);
79 
80  // Measure distance from middle to the furthest point of frustum slice
81  // Use to make a bounding sphere, but then convert into a bounding box
82  aabb[i] = glm::vec3(glm::distance(glm::vec3(maxCoord), middle[i]));
83  }

87  const glm::mat4 sunInverse = glm::inverse(sunModelMatrix);

89  auto & cam = cameraComponent->m_cameras[i];
90  auto & camData = *cam.get();
91  const glm::vec3 volumeUnitSize = (aabb[i] - -aabb[i]) / m_frameData->shadowData->shadowSize;
92  const glm::vec3 frustumpos = glm::vec3(sunModelMatrix * CamInv * glm::vec4(middle[i], 1.0f));
93  const glm::vec3 clampedPos = glm::floor((frustumpos + (volumeUnitSize / 2.0f)) / volumeUnitSize) * volumeUnitSize;
94  const glm::vec3 newMin = -aabb[i] + clampedPos;
95  const glm::vec3 newMax = aabb[i] + clampedPos;
96  const float l = newMin.x, r = newMax.x, b = newMax.y, t = newMin.y, n = -newMin.z, f = -newMax.z;
97  const glm::mat4 pMatrix = glm::ortho(l, r, b, t, n, f);
98  const glm::mat4 pvMatrix = pMatrix * sunModelMatrix;
99  const glm::vec4 v_near = CamP * glm::vec4(0, 0, cascadeEnd[i], 1.0f);
100  const glm::vec4 v_far = CamP * glm::vec4(0, 0, cascadeEnd[i + 1], 1.0f);
101  camData.Dimensions = glm::ivec2((int)m_frameData->shadowData->shadowSize);
102  camData.FOV = 90.0f;
103  camData.NearPlane = v_near.z;
104  camData.FarPlane = v_far.z;
105  auto pos = CamInv * glm::vec4(0, 0, -v_far.z / 2.0f, 1.0f);
106  pos /= pos.w;
107  camData.EyePosition = glm::vec3(pos);
108  camData.pMatrix = pMatrix;
109  camData.pMatrixInverse = glm::inverse(pMatrix);
110  camData.vMatrix = sunModelMatrix;
111  camData.vMatrixInverse = sunInverse;
112  camData.pvMatrix = pMatrix * sunModelMatrix;
113  cam.updateFrustum();
114  if (cam.getEnabled()) {
115  lightComponent->m_pvMatrices[i] = pvMatrix;
116  lightComponent->m_cascadeEnds[i] = v_far.z;
117  }
118  }
119  for (int x = 0; x < NUM_CASCADES; ++x) {
120  m_frameData->lightBuffer[index].lightVP[x] = lightComponent->m_pvMatrices[x];
121  m_frameData->lightBuffer[index].CascadeEndClipSpace[x] = lightComponent->m_cascadeEnds[x];
122  }
123  }
124 
125  // Sync Color Attributes
126  if (colorComponent) {
127  m_frameData->lightBuffer[index].LightColor = colorComponent->m_color;
128  m_frameData->lightBuffer[index].LightIntensity = colorComponent->m_intensity;
129  }
130

134  const auto matRot = glm::mat4_cast(orientation);

137  }
