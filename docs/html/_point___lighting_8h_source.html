134  inline virtual void updateComponents(const float & deltaTime, const std::vector< std::vector<BaseECSComponent*> > & components) override {
135  // Accumulate Light Data
136  std::vector<GLint> lightIndices, shadowIndices;
137  PriorityList<float, std::pair<LightPoint_Component*, LightPointShadow_Component*>, std::less<float>> oldest;
138  for each (const auto & componentParam in components) {
139  LightPoint_Component * lightComponent = (LightPoint_Component*)componentParam[0];
140  LightPointShadow_Component * shadowComponent = (LightPointShadow_Component*)componentParam[1];
141  Transform_Component * transformComponent = (Transform_Component*)componentParam[2];
142  const auto & index = *lightComponent->m_lightIndex;
143 
144  // Sync Transform Attributes
145  if (transformComponent) {
146  const auto & position = transformComponent->m_transform.m_position;
147  lightComponent->m_position = position;
148  const glm::mat4 trans = glm::translate(glm::mat4(1.0f), position);
149  const glm::mat4 scl = glm::scale(glm::mat4(1.0f), glm::vec3(lightComponent->m_radius*lightComponent->m_radius)*1.1f);
150  m_lightBuffer[index].mMatrix = (trans)* scl;
151 
152  if (shadowComponent) {
153  const auto & shadowIndex = *shadowComponent->m_shadowIndex;
154  m_shadowBuffer[shadowIndex].lightV = glm::translate(glm::mat4(1.0f), -position);
155  glm::mat4 rotMats[6];
156  const glm::mat4 pMatrix = glm::perspective(glm::radians(90.0f), 1.0f, 0.01f, lightComponent->m_radius*lightComponent->m_radius);
157  rotMats[0] = pMatrix * glm::lookAt(position, position + glm::vec3(1, 0, 0), glm::vec3(0, -1, 0));
158  rotMats[1] = pMatrix * glm::lookAt(position, position + glm::vec3(-1, 0, 0), glm::vec3(0, -1, 0));
159  rotMats[2] = pMatrix * glm::lookAt(position, position + glm::vec3(0, 1, 0), glm::vec3(0, 0, 1));
160  rotMats[3] = pMatrix * glm::lookAt(position, position + glm::vec3(0, -1, 0), glm::vec3(0, 0, -1));
161  rotMats[4] = pMatrix * glm::lookAt(position, position + glm::vec3(0, 0, 1), glm::vec3(0, -1, 0));
162  rotMats[5] = pMatrix * glm::lookAt(position, position + glm::vec3(0, 0, -1), glm::vec3(0, -1, 0));
163  for (int x = 0; x < 6; ++x) {
164  m_shadowBuffer[shadowIndex].lightPV[x] = rotMats[x];
165  m_shadowBuffer[shadowIndex].inversePV[x] = glm::inverse(rotMats[x]);
166  }
167  }
168  }
169 
170  // Update Buffer Attributes
171  m_lightBuffer[index].LightColor = lightComponent->m_color;
172  m_lightBuffer[index].LightPosition = lightComponent->m_position;
173  m_lightBuffer[index].LightIntensity = lightComponent->m_intensity;
174  m_lightBuffer[index].LightRadius = lightComponent->m_radius;
175  lightIndices.push_back((GLuint)index);
176  if (shadowComponent) {
177  shadowIndices.push_back((GLint)*shadowComponent->m_shadowIndex);
178  oldest.insert(shadowComponent->m_updateTime, std::make_pair(lightComponent, shadowComponent));
179  }
180  else
181  shadowIndices.push_back(-1);
182  }
183

185  const size_t & lightSize = lightIndices.size();
186  m_visLights.write(0, sizeof(GLuint) * lightSize, lightIndices.data());
187  m_visShadows.write(0, sizeof(GLuint) * shadowIndices.size(), shadowIndices.data());

189  m_shadowsToUpdate = PQtoVector(oldest);
190  }

263  PriorityList<float, std::pair<LightPoint_Component*, LightPointShadow_Component*>, std::greater<float>> m_closest(m_updateQuality / 2);
264  std::vector<std::pair<LightPoint_Component*, LightPointShadow_Component*>> outList;
