97  inline void expandToFit(const GLsizeiptr & offset, const GLsizeiptr & size) {
98  if (offset + size > m_maxCapacity) {
99  // Create new buffer large enough to fit old data + new data
100  const GLsizeiptr oldSize = m_maxCapacity;
101  m_maxCapacity += offset + (size * 2);
102 
103  for (int x = 0; x < 3; ++x) {
104  // Wait for this buffer in particular
105  if (m_fence[x] != nullptr)
106  while (1) {
107  GLenum waitReturn = glClientWaitSync(m_fence[x], GL_SYNC_FLUSH_COMMANDS_BIT, 1);
108  if (waitReturn == GL_SIGNALED || waitReturn == GL_ALREADY_SIGNALED || waitReturn == GL_CONDITION_SATISFIED) {
109  glDeleteSync(m_fence[x]);
110  m_fence[x] = nullptr;
111  break;
112  }
113  }
114 
115  // Create new buffer
116  GLuint newBuffer = 0;
117  glCreateBuffers(1, &newBuffer);
118  glNamedBufferStorage(newBuffer, m_maxCapacity, 0, GL_DYNAMIC_STORAGE_BIT | m_mapFlags);
119 
120  // Copy old buffer
121  if (oldSize)
122  glCopyNamedBufferSubData(m_bufferID[x], newBuffer, 0, 0, oldSize);
123 
124  // Delete old buffer
125  glUnmapNamedBuffer(m_bufferID[x]);
126  glDeleteBuffers(1, &m_bufferID[x]);
127 
128  // Migrate new buffer
129  m_bufferID[x] = newBuffer;
130  m_bufferPtr[x] = glMapNamedBufferRange(m_bufferID[x], 0, m_maxCapacity, m_mapFlags);
131  }
132  }
