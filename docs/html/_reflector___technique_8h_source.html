150  inline void updateReflectors(const float & deltaTime) {
151  if (m_reflectorsToUpdate->size()) {
152  // Copy the list of reflectors, because nested rendering may change the vector!
153  auto copy = *m_reflectorsToUpdate;
154  for (auto &[time, reflector, viewport] : copy) {
155  for (int x = 0; x < 6; ++x) {
156  // Set view-specific camera data
157  viewport->m_camera->m_cameraBuffer->beginWriting();
158  viewport->m_camera->m_cameraBuffer->replace(reflector->m_cameras[x]);
159  viewport->m_camera->m_cameraBuffer->pushChanges();
160 
161  // Render Graphics Pipeline
162  constexpr const unsigned int flags = Graphics_Technique::GEOMETRY | Graphics_Technique::PRIMARY_LIGHTING | Graphics_Technique::SECONDARY_LIGHTING;
163  m_engine->getModule_Graphics().render(deltaTime, viewport->m_camera, flags);
164  viewport->m_camera->setFOV(360.0f);
165  viewport->m_camera->m_cameraBuffer->endWriting();
166 
167  // Copy lighting frame into cube-face
168  viewport->m_camera->m_gfxFBOS->bindForReading("LIGHTING", 0);
169  m_envmapFBO.bindForWriting();
170  m_shaderCopy->bind();
171  m_shaderCopy->setUniform(0, x + reflector->m_cubeSpot);
172  m_indirectQuad.bindBuffer(GL_DRAW_INDIRECT_BUFFER);
173  glBindVertexArray(m_shapeQuad->m_vaoID);
174  glDrawArraysIndirect(GL_TRIANGLES, 0);
175  }
176 
177  // Once cubemap is generated, convolute it
178  m_shaderConvolute->bind();
179  m_shaderConvolute->setUniform(0, reflector->m_cubeSpot);
180  m_envmapFBO.bindForReading();
181  m_indirectQuad6Faces.bindBuffer(GL_DRAW_INDIRECT_BUFFER);
182  for (unsigned int r = 1; r < 6; ++r) {
183  // Ensure we are writing to MIP level r
184  const unsigned int write_size = (unsigned int)std::max(1.0f, (floor(m_envmapSize / pow(2.0f, (float)r))));
185  glViewport(0, 0, write_size, write_size);
186  m_shaderConvolute->setUniform(1, (float)r / 5.0f);
187  glNamedFramebufferTexture(m_envmapFBO.m_fboID, GL_COLOR_ATTACHMENT0, m_envmapFBO.m_textureID, r);
188 
189  // Ensure we are reading from MIP level r - 1
190  glTextureParameteri(m_envmapFBO.m_textureID, GL_TEXTURE_BASE_LEVEL, r - 1);
191  glTextureParameteri(m_envmapFBO.m_textureID, GL_TEXTURE_MAX_LEVEL, r - 1);
192 
193  // Convolute the 6 faces for this roughness level (RENDERS 6 TIMES)
194  glDrawArraysIndirect(GL_TRIANGLES, 0);
195  }
196 
197  // Reset texture, so it can be used for other component reflections
198  glTextureParameteri(m_envmapFBO.m_textureID, GL_TEXTURE_BASE_LEVEL, 0);
199  glTextureParameteri(m_envmapFBO.m_textureID, GL_TEXTURE_MAX_LEVEL, 5);
200  glNamedFramebufferTexture(m_envmapFBO.m_fboID, GL_COLOR_ATTACHMENT0, m_envmapFBO.m_textureID, 0);
201  reflector->m_updateTime = m_engine->getTime();
202  Shader::Release();
203  reflector->m_sceneOutOfDate = false;
204  }
205 
206  m_reflectorsToUpdate->clear();
207  m_buffers->envmapOutOfDate = false;
208  }

213  inline void renderReflectors(const float & deltaTime, const std::shared_ptr<Viewport> & viewport, const size_t & visibilityIndex) {
214  if (m_visibility->viewInfo[visibilityIndex].visLightCount) {
215  glEnable(GL_STENCIL_TEST);
216  glEnable(GL_BLEND);
217  glBlendEquation(GL_FUNC_ADD);
218  glBlendFunc(GL_ONE, GL_ONE);
219 
220  // Draw only into depth-stencil buffer
221  m_shaderStencil->bind(); // Shader (reflector)
222  viewport->m_gfxFBOS->bindForWriting("REFLECTION"); // Ensure writing to reflection FBO
223  viewport->m_gfxFBOS->bindForReading("GEOMETRY", 0); // Read from Geometry FBO
224  glBindTextureUnit(4, m_envmapFBO.m_textureID); // Reflection map (environment texture)
225  m_visibility->viewInfo[visibilityIndex].visLights.bindBufferBase(GL_SHADER_STORAGE_BUFFER, 3); // SSBO visible light indices
226  m_buffers->reflectorBuffer.bindBufferBase(GL_SHADER_STORAGE_BUFFER, 8); // Reflection buffer
227  m_visibility->viewInfo[visibilityIndex].indirectCube.bindBuffer(GL_DRAW_INDIRECT_BUFFER); // Draw call buffer
228  glBindVertexArray(m_shapeCube->m_vaoID); // Quad VAO
229  glEnable(GL_DEPTH_TEST);
230  glDisable(GL_CULL_FACE);
231  glClear(GL_STENCIL_BUFFER_BIT);
232  glStencilFunc(GL_ALWAYS, 0, 0);
233  glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
234  glDepthMask(GL_FALSE);
235  glFrontFace(GL_CW);
236  glDrawArraysIndirect(GL_TRIANGLES, 0);
237  glFrontFace(GL_CCW);
238 
239  // Now draw into color buffers
240  m_shaderLighting->bind();
241  glDisable(GL_DEPTH_TEST);
242  glEnable(GL_CULL_FACE);
243  glDisable(GL_BLEND);
244  glCullFace(GL_FRONT);
245  glStencilFunc(GL_NOTEQUAL, 0, 0xFF);
246  glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
247  glDrawArraysIndirect(GL_TRIANGLES, 0);
248 
249  glDepthMask(GL_TRUE);
250  glCullFace(GL_BACK);
251  glBlendFunc(GL_ONE, GL_ZERO);
252  glDisable(GL_STENCIL_TEST);
253  }
