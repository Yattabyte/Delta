93  // Exit Early
94  if (m_enabled && m_frameData->viewInfo.size() && m_shapeCube->existsYet() && m_shaderLighting->existsYet() && m_shaderStencil->existsYet()) {
95  if (m_drawIndex >= m_drawData.size())
96  m_drawData.resize(m_drawIndex + 1);

98  auto &camBufferIndex = drawBuffer.bufferCamIndex;
99  auto &lightBufferIndex = drawBuffer.visLights;
100  camBufferIndex.beginWriting();
101  lightBufferIndex.beginWriting();
102  drawBuffer.indirectShape.beginWriting();
103 
104  // Accumulate all visibility info for the cameras passed in
105  std::vector<glm::ivec2> camIndices;
106  std::vector<GLint> lightIndices;
107  for (auto &[camIndex, layer] : perspectives) {
108  const std::vector<glm::ivec2> tempIndices(m_frameData->viewInfo[camIndex].lightIndices.size(), { camIndex, layer });
109  camIndices.insert(camIndices.end(), tempIndices.begin(), tempIndices.end());
110  lightIndices.insert(lightIndices.end(), m_frameData->viewInfo[camIndex].lightIndices.begin(), m_frameData->viewInfo[camIndex].lightIndices.end());
111  }
112 
113  if (lightIndices.size()) {
114  // Write accumulated data
115  camBufferIndex.write(0, sizeof(glm::ivec2) * camIndices.size(), camIndices.data());
116  lightBufferIndex.write(0, sizeof(GLuint) * lightIndices.size(), lightIndices.data());
117  const GLuint dataShape[] = { (GLuint)m_shapeCube->getSize(), (GLuint)lightIndices.size(), 0,0 };
118  drawBuffer.indirectShape.write(0, sizeof(GLuint) * 4, &dataShape);
119 
120  // Render lights
121  renderReflectors(deltaTime, viewport);
122 
123  m_drawIndex++;
124  }
125  }

134  inline void updateReflectors(const float & deltaTime) {
135  auto clientTime = m_engine->getTime();
136  if (m_frameData->reflectorsToUpdate.size()) {
137  m_viewport->resize(m_frameData->envmapSize, (int)m_frameData->reflectorLayers);
138  m_indirectQuad.beginWriting();
139  m_indirectQuadConvolute.beginWriting();
140 
141  // Accumulate Perspective Data
142  std::vector<std::pair<int, int>> perspectives;
143  perspectives.reserve(m_frameData->reflectorsToUpdate.size());
144  for (auto &[importance, time, reflectorSpot, camera] : m_frameData->reflectorsToUpdate) {
145  // Accumulate all visibility info for the cameras passed in
146  int visibilityIndex = 0;
147  bool found = false;
148  for (int y = 0; y < m_sceneCameras->size(); ++y)
149  if (m_sceneCameras->at(y) == camera) {
150  visibilityIndex = y;
151  found = true;
152  break;
153  }
154  if (found)
155  perspectives.push_back({ visibilityIndex, reflectorSpot });
156  *time = clientTime;
157  camera->setEnabled(false);
158  }
159  const auto instanceCount = (GLuint)perspectives.size(), convoluteCount = (GLuint)m_frameData->reflectorLayers;
160  m_indirectQuad.write(sizeof(GLuint), sizeof(GLuint), &instanceCount);
161  m_indirectQuadConvolute.write(sizeof(GLuint), sizeof(GLuint), &convoluteCount);
162 
163  // Update all reflectors at once
164  m_engine->getModule_Graphics().renderScene(deltaTime, m_viewport, perspectives, Graphics_Technique::GEOMETRY | Graphics_Technique::PRIMARY_LIGHTING | Graphics_Technique::SECONDARY_LIGHTING);
165 
166  // Copy all lighting results into cube faces, generating cubemaps
167  m_viewport->m_gfxFBOS->bindForReading("LIGHTING", 0);
168  m_frameData->envmapFBO.bindForWriting(0);
169  m_shaderCopy->bind();
170  m_indirectQuad.bindBuffer(GL_DRAW_INDIRECT_BUFFER);
171  glDisable(GL_DEPTH_TEST);
172  glDisable(GL_BLEND);
173  glDisable(GL_STENCIL_TEST);
174  glBindVertexArray(m_shapeQuad->m_vaoID);
175  glDrawArraysIndirect(GL_TRIANGLES, 0);
176 
177  // Convolute all completed cubemaps, not just what was done this frame
178  m_shaderConvolute->bind();
179  m_indirectQuadConvolute.bindBuffer(GL_DRAW_INDIRECT_BUFFER);
180  m_frameData->envmapFBO.bindForReading(0);
181  for (unsigned int r = 1; r < 6; ++r) {
182  // Ensure we are writing to MIP level r
183  const unsigned int write_size = (unsigned int)std::max(1.0f, (floor((float)m_frameData->envmapSize.x / pow(2.0f, (float)r))));
184  glViewport(0, 0, write_size, write_size);
185  m_shaderConvolute->setUniform(1, (float)r / 5.0f);
186  m_frameData->envmapFBO.bindForWriting(r);
187 
188  // Convolute the 6 faces for this roughness level
189  glDrawArraysIndirect(GL_TRIANGLES, 0);
190  }
191 
192  m_frameData->reflectorsToUpdate.clear();
193  }
