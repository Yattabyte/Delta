109  didAnything = true;
110  }
111  reflector->m_outOfDate = false;
112  for (int x = 0; x < 6; ++x) {

114  graphics.frameTick(deltaTime);
115 
116  // Copy lighting frame into cube-face
117  m_lightingFBO->bindForReading();
118  m_envmapFBO.bindForWriting();
119  m_shaderCopy->bind();

121  m_renderState->m_indirectQuad.bindBuffer(GL_DRAW_INDIRECT_BUFFER);
122  glBindVertexArray(m_shapeQuad->m_vaoID);
123  glDrawArraysIndirect(GL_TRIANGLES, 0);
124  }
125  // Once cubemap is generated, convolute it
126  m_shaderConvolute->bind();
127  m_shaderConvolute->setUniform(0, reflector->m_cubeSpot);
128  m_envmapFBO.bindForReading();

130  for (unsigned int r = 1; r < 6; ++r) {
131  // Ensure we are writing to MIP level r
132  const unsigned int write_size = (unsigned int)std::max(1.0f, (floor(m_renderState->m_envmapSize / pow(2.0f, (float)r))));
133  glViewport(0, 0, write_size, write_size);
134  m_shaderConvolute->setUniform(1, (float)r / 5.0f);
135  glNamedFramebufferTexture(m_envmapFBO.m_fboID, GL_COLOR_ATTACHMENT0, m_envmapFBO.m_textureID, r);
136

138  glTextureParameteri(m_envmapFBO.m_textureID, GL_TEXTURE_BASE_LEVEL, r - 1);
139  glTextureParameteri(m_envmapFBO.m_textureID, GL_TEXTURE_MAX_LEVEL, r - 1);
140 
141  // Convolute the 6 faces for this roughness level (RENDERS 6 TIMES)
142  glDrawArraysIndirect(GL_TRIANGLES, 0);
143  }
