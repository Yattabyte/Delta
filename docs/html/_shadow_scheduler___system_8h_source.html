44  const auto clientTime = m_engine->getTime();

52  auto tryToAddShadow = [&shadows, &maxShadows, &clientPosition, &clientFarPlane, &clientTime](const int& shadowSpot, Camera* cb, float* updateTime) {
53  const float linDist = glm::distance(clientPosition, cb->getFrustumCenter()) / clientFarPlane;
54  const float importance_distance = 1.0f - (linDist * linDist);
55 
56  const float linTime = (clientTime - *updateTime) / 5.0f;
57  const float importance_time = linTime * linTime;
58 
59  const float importance = importance_distance + importance_time * (1.0f - importance_distance);
60  bool didAnything = false;
61  // Try to find the oldest components
62  for (int x = 0; x < shadows.size(); ++x) {
63  auto& [oldImportance, oldTime, oldShadowSpot, oldCamera] = shadows[x];
64  if ((oldShadowSpot != -1 && importance > oldImportance)) {
65  // Expand container by one
66  shadows.resize(shadows.size() + 1);
67  // Shuffle next elements down
68  for (auto y = shadows.size() - 1ull; y > x; --y)
69  shadows[y] = shadows[y - 1ull];
70  oldImportance = importance;
71  oldTime = updateTime;
72  oldShadowSpot = shadowSpot;
73  oldCamera = cb;
74  didAnything = true;
75  break;
76  }
77  }
78  if (!didAnything && shadows.size() < maxShadows)
79  shadows.push_back({ importance, updateTime, shadowSpot, cb });
80  if (shadows.size() > maxShadows)
81  shadows.resize(maxShadows);
82  };
83  // Set appropriate shadow spot
84  shadowComponent->m_shadowSpot = cameraCount;
85 
86  if (camSingle) {
87  camSingle->m_camera.setEnabled(false);
88  tryToAddShadow(shadowComponent->m_shadowSpot, &(camSingle->m_camera), &camSingle->m_updateTime);
89  cameraCount++;
90  }
91  else if (camArrays) {
92  camArrays->m_updateTimes.resize(camArrays->m_cameras.size());
93  for (int x = 0; x < camArrays->m_cameras.size(); ++x) {
94  camArrays->m_cameras[x].setEnabled(false);

98  }
99  }

104
