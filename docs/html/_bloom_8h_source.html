98 
99  // Blur bright regions

101  glNamedFramebufferTexture(m_fbo_GB, GL_COLOR_ATTACHMENT0, m_textureIDS_GB[0], 0);
102  glNamedFramebufferTexture(m_fbo_GB, GL_COLOR_ATTACHMENT1, m_textureIDS_GB[1], 0);
103 
104  // Read from desired texture, blur into this frame buffer
105  bool horizontal = false;
106  glBindTextureUnit(0, m_textureID);
107  glBindTextureUnit(1, m_textureIDS_GB[0]);
108  glDrawBuffer(GL_COLOR_ATTACHMENT0 + horizontal);
109  m_shaderGB->bind();
110  m_shaderGB->setUniform(0, horizontal);
111  m_shaderGB->setUniform(1, glm::vec2(m_renderSize));
112 
113  glBindVertexArray(m_shapeQuad->m_vaoID);
114  m_quadIndirectBuffer.bindBuffer(GL_DRAW_INDIRECT_BUFFER);
115  glDrawArraysIndirect(GL_TRIANGLES, 0);
116 
117  // Blur remainder of the times minus 1
118  glBindTextureUnit(0, m_textureIDS_GB[1]);
119  for (int i = 1; i < m_bloomStrength - 1; i++) {
120  horizontal = !horizontal;
121  glDrawBuffer(GL_COLOR_ATTACHMENT0 + horizontal);
122  m_shaderGB->setUniform(0, horizontal);
123  glDrawArraysIndirect(GL_TRIANGLES, 0);
124  }
125 
126  // Last blur back into desired texture
127  horizontal = !horizontal;
128  glNamedFramebufferTexture(m_fbo_GB, GL_COLOR_ATTACHMENT0, m_textureID, 0);
129  glDrawBuffer(GL_COLOR_ATTACHMENT0 + horizontal);
130  m_shaderGB->setUniform(0, horizontal);
131  glDrawArraysIndirect(GL_TRIANGLES, 0);
132 
133  // Re-attach our bloom texture (was detached to allow for convolution)
134  glNamedFramebufferTexture(m_fboID, GL_COLOR_ATTACHMENT0, m_textureID, 0);
135 
136  // Copy to lighting buffer
137  glEnable(GL_BLEND);

139  glBlendFunc(GL_ONE, GL_ONE);

145  glDrawArraysIndirect(GL_TRIANGLES, 0);
146  glDisable(GL_BLEND);
147  glBindTextureUnit(0, m_lightingTexID);
148  }
