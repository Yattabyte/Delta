84  inline virtual void keyboardAction(const KeyboardEvent & keyboardEvent) override {
85  if (m_edit) {
86  // Check for a text stream
87  if (auto character = keyboardEvent.getChar()) {
88  setText(m_text.substr(0, m_caretIndex) + char(character) + m_text.substr(m_caretIndex, m_text.size()));
89  setCaret(m_caretIndex + 1);
90  enactCallback(on_text_change);
91  }
92  // Otherwise, check keyboard states
93  else {
94  if (keyboardEvent.getState(KeyboardEvent::ENTER) || keyboardEvent.getState(KeyboardEvent::ESCAPE))
95  m_edit = false;
96  else if (keyboardEvent.getState(KeyboardEvent::BACKSPACE)) {
97  if (m_caretIndex > 0) {
98  setText(m_text.substr(0, m_caretIndex - 1) + m_text.substr(m_caretIndex, m_text.size()));
99  setCaret(m_caretIndex - 1);
100  enactCallback(on_text_change);
101  }
102  }
103  else if (keyboardEvent.getState(KeyboardEvent::DEL)) {
104  if (m_caretIndex + 1 <= m_text.size()) {
105  setText(m_text.substr(0, m_caretIndex) + m_text.substr(m_caretIndex + 1, m_text.size()));
106  enactCallback(on_text_change);
107  }
108  }
109  else if (keyboardEvent.getState(KeyboardEvent::LEFT))
110  setCaret(m_caretIndex - 1);
111  else if (keyboardEvent.getState(KeyboardEvent::RIGHT))
112  setCaret(m_caretIndex + 1);
113  }
114  }
