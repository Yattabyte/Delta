115  inline void tryInsertModel(const Shared_Model & model) {
116  if (m_modelMap.find(model) == m_modelMap.end()) {
117  // Prop hasn't been uploaded yet
118  const size_t arraySize = model->m_data.m_vertices.size() * sizeof(SingleVertex);
119  // Check if we can fit the desired data
120  waitOnFence();
121  tryToExpand(arraySize);
122  auto offset = (GLuint)(m_currentSize / sizeof(SingleVertex));
123  auto count = (GLuint)(arraySize / sizeof(SingleVertex));
124 
125  // Upload vertex data
126  glNamedBufferSubData(m_vboID, m_currentSize, arraySize, &model->m_data.m_vertices[0]);
127  m_currentSize += arraySize;
128 
129  // Prepare fence
130  m_fence = glFenceSync(GL_SYNC_GPU_COMMANDS_COMPLETE, 0);
131  m_modelMap[model] = {offset, count};
132  }
133  }
135  inline void waitOnFence() {
136  if (m_fence) {
137  // Wait for data fence to be passed
138  GLenum state = GL_UNSIGNALED;
139  while (state != GL_SIGNALED && state != GL_ALREADY_SIGNALED && state != GL_CONDITION_SATISFIED)
140  state = glClientWaitSync(m_fence, GL_SYNC_FLUSH_COMMANDS_BIT, 1);
141  glDeleteSync(m_fence);
142  m_fence = nullptr;
143  }
144  }
147  inline void tryToExpand(const size_t & arraySize) {
148  if (m_currentSize + arraySize > m_maxCapacity) {
149  // Create new set of VBO's large enough to fit old data + desired data
150  m_maxCapacity += arraySize * 2;
151 
152  // Create the new VBO's
153  GLuint newVBOID = 0;
154  glCreateBuffers(1, &newVBOID);
155  glNamedBufferStorage(newVBOID, m_maxCapacity, 0, GL_DYNAMIC_STORAGE_BIT);
156 
157  // Copy old VBO's
158  glCopyNamedBufferSubData(m_vboID, newVBOID, 0, 0, m_currentSize);
159 
160  // Delete the old VBO's
161  glDeleteBuffers(1, &m_vboID);
162 
163  // Overwrite old VBO ID's
164  m_vboID = newVBOID;
165 
166  // Assign VAO to new VBO's
167  glVertexArrayVertexBuffer(m_vaoID, 0, m_vboID, 0, sizeof(SingleVertex));
168  }

172  inline void tryInsertMaterial(const Shared_Material & material) {
173  if (m_materialMap.find(material) == m_materialMap.end()) {
174  // Get spot in the material array
175  const auto imageCount = (GLsizei)((material->m_textures.size() / MAX_PHYSICAL_IMAGES) * MAX_DIGITAL_IMAGES);
176  auto materialID = (GLuint)m_matCount;
177  m_matCount += imageCount;
178  if (m_matCount >= m_maxTextureLayers)
179  m_engine->getManager_Messages().error("Out of room for more materials!");
180 
181  // Upload material data
182  GLuint pboID = 0;
183  glCreateBuffers(1, &pboID);
184  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, pboID);
185  glNamedBufferStorage(pboID, m_materialSize * m_materialSize * imageCount * 4, material->m_materialData, GL_DYNAMIC_STORAGE_BIT);
186  for (int x = 0; x < m_maxMips; ++x) {
187  const GLsizei mipsize = (GLsizei)std::max(1.0f, (floor(m_materialSize / pow(2.0f, (float)x))));
188  glTexturePageCommitmentEXT(m_matID, x, 0, 0, materialID, mipsize, mipsize, imageCount, GL_TRUE);
189  }
190  glTextureSubImage3D(m_matID, 0, 0, 0, materialID, m_materialSize, m_materialSize, imageCount, GL_RGBA, GL_UNSIGNED_BYTE, (void *)0);
191  glGenerateTextureMipmap(m_matID);
192  glBindBuffer(GL_PIXEL_UNPACK_BUFFER, 0);
193  glDeleteBuffers(1, &pboID);
194  m_materialMap[material] = materialID;
195  }
196  }
