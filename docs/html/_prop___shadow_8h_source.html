100  // Accumulate draw parameter information per model
101  std::vector<glm::ivec4> cullingDrawData, renderingDrawData;
102  std::vector<GLuint> visibleIndices;
103  std::vector<int> skeletonData;
104  const bool renderStatic = m_flags & RenderStatic;
105  const bool renderDynamic = m_flags & RenderDynamic;
106  for each (const auto & componentParam in components) {
107  Prop_Component * propComponent = (Prop_Component*)componentParam[0];
108  Skeleton_Component * skeletonComponent = (Skeleton_Component*)componentParam[1];
109  if (!propComponent->m_model->existsYet())
110  continue; // Skip if prop isn't ready
111  if (skeletonComponent == nullptr && renderDynamic && !renderStatic)
112  continue; // Skip if we aren't supporting static props
113  else if (skeletonComponent && renderStatic && !renderDynamic)
114  continue; // Skip if we aren't supporting dynamic props
115 
116  const auto & offset = propComponent->m_model->m_offset;
117  const auto & count = propComponent->m_model->m_count;
118  const auto & index = *propComponent->m_propBufferIndex;
119  visibleIndices.push_back((GLuint)index);
120  // Flag for occlusion culling if mesh complexity is high enough and if viewer is NOT within BSphere
121  if ((count >= 100) && propComponent->m_radius < glm::distance(propComponent->m_position, m_lightPos)) {
122  // Allow
123  cullingDrawData.push_back(glm::ivec4(36, m_instanceCount, 0, 1));
124  renderingDrawData.push_back(glm::ivec4(count, 0, offset, 1));
125  }
126  else {
127  // Skip occlusion culling
128  cullingDrawData.push_back(glm::ivec4(36, 0, 0, 1));
129  renderingDrawData.push_back(glm::ivec4(count, m_instanceCount, offset, 1));
130  }
131  skeletonData.push_back(skeletonComponent ? (GLint)*skeletonComponent->m_skeleBufferIndex : -1); // get skeleton ID if this entity has one
132  }
133 
134  // Update camera buffers
135  m_propCount = (GLsizei)visibleIndices.size();
136  m_bufferPropIndex.write(0, sizeof(GLuint) * m_propCount, visibleIndices.data());
137  m_bufferCulling.write(0, sizeof(glm::ivec4) * m_propCount, cullingDrawData.data());

139  m_bufferSkeletonIndex.write(0, sizeof(int) * m_propCount, skeletonData.data());
