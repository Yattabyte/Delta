88  inline void updateVisibility(const float & deltaTime, const std::vector<std::vector<BaseECSComponent*>> & components) {
89  // Accumulate draw parameter information per model
90  std::vector<glm::ivec4> cullingDrawData, renderingDrawData;
91  std::vector<GLuint> visibleIndices;
92  std::vector<int> skeletonData;
93  const bool renderStatic = m_flags & RenderStatic;
94  const bool renderDynamic = m_flags & RenderDynamic;
95  for each (const auto & componentParam in components) {
96  Prop_Component * propComponent = (Prop_Component*)componentParam[0];
97  Skeleton_Component * skeletonComponent = (Skeleton_Component*)componentParam[1];
98  if (!propComponent->m_model->existsYet())
99  continue; // Skip if prop isn't ready
100  if (renderDynamic && !renderStatic && propComponent->m_static)
101  continue; // Skip if we aren't supporting static props
102  else if (renderStatic && !renderDynamic && !propComponent->m_static)
103  continue; // Skip if we aren't supporting dynamic props
104 
105  const auto & offset = propComponent->m_model->m_offset;
106  const auto & count = propComponent->m_model->m_count;
107  const auto & index = *propComponent->m_modelBufferIndex;
108  visibleIndices.push_back((GLuint)index);
109  // Flag for occlusion culling if mesh complexity is high enough and if viewer is NOT within BSphere
110  if ((count >= 100) && propComponent->m_radius < glm::distance(propComponent->m_position, m_lightPos)) {
111  // Allow
112  cullingDrawData.push_back(glm::ivec4(36, m_instanceCount, 0, 1));
113  renderingDrawData.push_back(glm::ivec4(count, 0, offset, 1));
114  }
115  else {
116  // Skip occlusion culling
117  cullingDrawData.push_back(glm::ivec4(36, 0, 0, 1));
118  renderingDrawData.push_back(glm::ivec4(count, m_instanceCount, offset, 1));
119  }
120  skeletonData.push_back(skeletonComponent ? (GLint)*skeletonComponent->m_skeleBufferIndex : -1); // get skeleton ID if this entity has one
121  }
122 
123  // Update camera buffers
124  m_visProps = (GLsizei)visibleIndices.size();
125  m_bufferPropIndex.write(0, sizeof(GLuint) * m_visProps, visibleIndices.data());
126  m_bufferCulling.write(0, sizeof(glm::ivec4) * m_visProps, cullingDrawData.data());
127  m_bufferRender.write(0, sizeof(glm::ivec4) * m_visProps, renderingDrawData.data());
128  m_bufferSkeletonIndex.write(0, sizeof(int) * m_visProps, skeletonData.data());
129  }
